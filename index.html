            Python样例1（绘图）

            xxxxxxxxxx
            from matplotlib import pyplot as plt
            import numpy as np
            

            # 生成原始三维数据点
            x, y, scale = np.random.randn(3, 100)
            fig, ax = plt.subplots()
            

            # 三维数据平面呈现——xy坐标+颜色和大小
            ax.scatter(x=x, y=y, c=scale, s=np.abs(scale)*500)
            ax.set(title="标题设置")
            plt.show()
